Your team received a messy dataset with user contact information. You need to use regular expressions to clean and validate the data. Here are some sample entries:

# Basic email pattern [A-Za-z0-9._%-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,} # Explanation: # [A-Za-z0-9._%-]+ : username part (letters, numbers, dots, underscores, %, -) # @ : literal @ symbol # [A-Za-z0-9.-]+ : domain name (letters, numbers, dots, hyphens) # \. : literal dot (escaped) # [A-Za-z]{2,4} : top-level domain (2-4 letters)

# Pattern for all phone formats shown (\([0-9]{3}\)\s?|[0-9]{3}[-.]?)[0-9]{3}[-.]?[0-9]{4} # Alternative - simpler approach matching each format separately: # Format 1: (555) 123-4567 \([0-9]{3}\)\s[0-9]{3}-[0-9]{4} # Format 2: 555.987.6543 [0-9]{3}\.[0-9]{3}\.[0-9]{4} # Format 3: 555-111-2222 [0-9]{3}-[0-9]{3}-[0-9]{4}

# ISO format: 2024-01-15 [0-9]{4}-[0-9]{2}-[0-9]{2} # US format: 01/15/2024 [0-9]{2}/[0-9]{2}/[0-9]{4} # Day-Month-Year: 15-Jan-2024 [0-9]{1,2}-(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)-[0-9]{4}

def normalize_scores(scores: list) -> list: """ Normalize a list of scores to a 0-100 scale based on min-max normalization. Parameters ---------- scores : list List of numerical scores to normalize Returns ------- list List of scores normalized to 0-100 scale Examples -------- >>> normalize_scores([50, 75, 100]) [0.0, 50.0, 100.0] """ if not scores: return [] if len(scores) == 1: return [50.0] # Single score defaults to middle min_score = min(scores) max_score = max(scores) # Handle case where all scores are identical if max_score == min_score: return [50.0] * len(scores) normalized = [] for score in scores: norm_score = 100 * (score - min_score) / (max_score - min_score) normalized.append(norm_score) return normalized

Question 9: Advanced Function Features & Error Handling

Create a function called safe_divide_lists that:

  • Takes two lists and performs element-wise division
  • Accepts optional arguments for handling division by zero
  • Uses *args and **kwargs
  • Returns multiple values: (results, errors_encountered)
  • Has proper error handling with try/except
# Should work like this: results, errors = safe_divide_lists([10, 20, 30], [2, 0, 5], default_value='skip', precision=2) # Results: [5.0, 15.0], errors: ['Division by zero at index 1']

Solution:

def safe_divide_lists(numerators: list, denominators: list, default_value='skip', precision=None, *args, **kwargs) -> tuple: """ Perform element-wise division with error handling. Parameters ---------- numerators : list List of numbers to be divided denominators : list List of numbers to divide by default_value : str or float, optional How to handle division by zero, by default 'skip' precision : int, optional Decimal places to round results, by default None Returns ------- tuple (results_list, errors_list) """ results = [] errors = [] # Handle lists of different lengths max_length = min(len(numerators), len(denominators)) if len(numerators) != len(denominators): errors.append(f"Lists have different lengths, processing first {max_length} elements") # Process each pair for i in range(max_length): try: numerator = numerators[i] denominator = denominators[i] # Check for division by zero if denominator == 0: if default_value == 'skip': errors.append(f"Division by zero at index {i}") continue elif default_value == 'zero': result = 0 elif isinstance(default_value, (int, float)): result = default_value else: errors.append(f"Invalid default_value at index {i}") continue else: result = numerator / denominator # Apply precision if specified if precision is not None: result = round(result, precision) results.append(result) except TypeError as e: errors.append(f"Type error at index {i}: {str(e)}") except Exception as e: errors.append(f"Unexpected error at index {i}: {str(e)}") return results, errors

class gradeManager(): def __init__(self,student_grades=[]): self.grades=student_grades self.passing_grade=60 def add_grade(self,student,grade): if grade >= 0: self.grades.append((student,grade)) def get_average(self): total=0 for student,grade in self.grades: total=total+grade return total/len(self.grades) def get_passing_students(self): passing=[] for student,grade in self.grades: if grade>self.passing_grade: passing.append(student) return passing # Usage gm=gradeManager() gm.add_grade("Alice",85) gm.add_grade("Bob",-5) print("Average:",gm.get_average())

Your tasks:

  • Identify all bugs (logical, runtime, and edge cases)
  • Fix PEP 8 style violations
  • Add proper error handling and validation
  • Write comprehensive docstrings and tests
  • Discuss the mutable default argument problem

Solution:

Issues Identified:

  1. Mutable default argument: student_grades=[] creates shared list
  2. Class naming: Should be GradeManager (PascalCase)
  3. Division by zero: No check for empty grades list
  4. Logic error: Negative grades allowed, should validate range
  5. Passing grade logic: Uses > instead of >= (typically 60+ is passing)
  6. Style violations: Multiple PEP 8 spacing issues

Fixed Code:

from typing import List, Tuple, Optional class InvalidGradeError(Exception): """Raised when an invalid grade is provided.""" pass class GradeManager: """Manages student grades with validation and analysis capabilities.""" def __init__(self, student_grades: Optional[List[Tuple[str, float]]] = None, passing_grade: float = 60.0): """ Initialize GradeManager. Parameters ---------- student_grades : List[Tuple[str, float]], optional List of (student_name, grade) tuples, by default None passing_grade : float, optional Minimum grade considered passing, by default 60.0 """ # Fix mutable default argument issue self.grades = student_grades if student_grades is not None else [] self.passing_grade = passing_grade # Validate initial grades for student, grade in self.grades: self._validate_grade(grade) def _validate_grade(self, grade: float) -> None: """Validate that grade is in acceptable range.""" if not isinstance(grade, (int, float)): raise InvalidGradeError(f"Grade must be a number, got {type(grade)}") if not (0 <= grade <= 100): raise InvalidGradeError(f"Grade must be between 0 and 100, got {grade}") def add_grade(self, student: str, grade: float) -> None: """ Add a grade for a student. Parameters ---------- student : str Name of the student grade : float Grade to add (must be between 0 and 100) """ if not isinstance(student, str) or not student.strip(): raise ValueError("Student name must be a non-empty string") self._validate_grade(grade) self.grades.append((student, grade)) def get_average(self) -> float: """ Calculate the average grade. Returns ------- float Average grade rounded to 2 decimal places """ if not self.grades: raise ValueError("Cannot calculate average: no grades available") total = sum(grade for _, grade in self.grades) return round(total / len(self.grades), 2) def get_passing_students(self) -> List[str]: """ Get list of students with passing grades. Returns ------- List[str] Names of students with grades >= passing_grade """ passing = [] for student, grade in self.grades: if grade >= self.passing_grade: # Fixed: was > instead of >= passing.append(student) return passing def __str__(self) -> str: """String representation of GradeManager.""" if not self.grades: return "GradeManager: No grades recorded" avg = self.get_average() if self.grades else 0 passing_count = len(self.get_passing_students()) return (f"GradeManager: {len(self.grades)} students, " f"Average: {avg}, Passing: {passing_count}") # Test the fixed implementation def test_grade_manager(): """Test the fixed GradeManager.""" # Test normal functionality gm = GradeManager() gm.add_grade("Alice", 85) gm.add_grade("Bob", 75) gm.add_grade("Charlie", 55) # Failing grade print("Grade Manager:", gm) print("Average:", gm.get_average()) print("Passing students:", gm.get_passing_students()) # Test error handling try: gm.add_grade("Invalid", -5) except InvalidGradeError as e: print("Caught expected error:", e) # Test mutable default argument fix gm1 = GradeManager() gm2 = GradeManager() gm1.add_grade("Student1", 80) gm2.add_grade("Student2", 90) print("GM1 grades:", len(gm1.grades)) # Should be 1 print("GM2 grades:", len(gm2.grades)) # Should be 1 test_grade_manager()

Key Debugging Lessons:

  • Mutable Default Arguments: Never use [] or {} as default parameters
  • Edge Case Testing: Always test empty inputs, boundary conditions
  • Input Validation: Validate all inputs at entry points
  • Boundary Logic: Be careful with > vs >= comparisons
  • Comprehensive Testing: Test normal cases, edge cases, and error conditions
